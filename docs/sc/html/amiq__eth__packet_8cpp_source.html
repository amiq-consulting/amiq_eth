00030 void amiq_eth_do_unpack(amiq_eth_packer& packer, sc_bv_base& value) {
00031         value = 0;
00032         for (int i = 0; i < value.length(); i++) {
00033                 bool bit;
00034                 packer >> bit;
00035                 value = (value << 1) | bit;
00036         }
00037 }

00039 void amiq_eth_do_pack(amiq_eth_packer& packer, const sc_bv_base& value) {
00040         uint64 uint64_value = value.to_uint64();
00041         for (int i = value.length() - 1; i >= 0; i--) {
00042                 bool bit = (bool) ((uint64_value >> i) & 1);
00043                 packer << bit;
00044         }
